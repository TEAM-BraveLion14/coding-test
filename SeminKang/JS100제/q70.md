### 내 코드
```js
function matrixMulti(a, b){
    const result = [];

    if (a.filter(e => {
        return e.length === b.length;
    })){
        a.forEach(e => {
            const resultEach = [];
            for (let i = 0; i < e.length; i++){
                let j = 0;
                let res = 0;
                while (j < e.length){
                    res += e[j] * b[j][i];
                    j++;
                }
                resultEach.push(res);
            }
            result.push(resultEach);
        });

        return result;
    } else {
        return -1;
    }
}
```
